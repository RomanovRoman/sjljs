                case 'Object':
                    sjl.extend(true, self, attr, true);
                    break;

                case 'String':

                    if (arguments.length >= 2) {

                    // Is getter
                    else {
                        retVal = sjl.getValueFromObj(attr, self);

                    break;
            }

            return retVal;
        },
